% Define command for watermarking, with a DRAFT specific one
\usepackage{eso-pic}
\newcommand{\watermark}[1]
{
  \AddToShipoutPicture{%
    \definecolor{lightgray}{gray}{.85}
    \setlength{\unitlength}{1in}
    \put(2.5,3.5){%
      \rotatebox{45}{%
        \resizebox{4in}{1in}{%
          \textsf{\textcolor{lightgray}{#1}}
        }
      }
    }
  }
}
